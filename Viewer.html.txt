<body>
  <h1>PCCheck Results</h1>
  <div id="content">Loading...</div>

  <script>
    async function loadResults() {
      try {
        const r = await fetch('/Results.txt');
        const txt = await r.text();
        document.getElementById('content').innerHTML = '<pre>' + txt.replace(/</g,'&lt;') + '</pre>';
      } catch(e) {
        document.getElementById('content').innerText = 'Results.txt not found or server not running.';
      }
    }
    loadResults();
  </script>
